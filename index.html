    kanjo

    Summarize AWS Billing

    Installation

    npm install kanjo

    Usage

    import Kanjo from 'kanjo';

const kanjo = new Kanjo({
  account: 'ACCOUNT_ID',
  bucket: 'BUCKET_NAME',
  region: 'us-east-1'
});

kanjo.fetch(2015, 9).then(report => {
  console.log(report);
});
